# Alexandria Seeks are automatically-generated mission sequences which
# take you all over some area of the galaxy searching for a
# book. There are six levels of difficulty of Alexandria Seeks
# depending on what you have done and what you have discovered.  Level
# 0 is for players that are still in their original ship, and level 5
# is for late-game missions.

# Level 5 Calculation: nominally, you can fight fleets of tier 2
# ships, you've met everyone, and you have progressed at least a
# little bit in all of the alien stories.

mission "Alexandria Seek: Level 5 Calculation"
	landing
	repeat
	invisible
	source
		not
			attributes uninhabited
	to offer
		has "library card"
		or
			"armament deterrence" > 50000
			"combat rating" > 10000
		has "license: Wanderer Military"
		has "license: Coalition"
		has "license: Remnant Capital"
		has "Discovered Korath Space: fail"
		has "First Contact: Quarg: declined"
		or
			has "Disabled Ka'het: declined"
			has "First Contact: Ka'het: Remnant 1B: offered"
			has "First Contact: Ka'het: Remnant 1B: active"
			has "First Contact: Ka'het: Quarg 2: offered"
			has "First Contact: Ka'het: Quarg 2: active"
	on offer
		"alexandria seeker level" >?= 5
		fail

mission "Alexandria Seek Near Paradise Planets: Start"
	job
	repeat
	name "Alexandria Seek: <planet>"
	description "Meet a Seeker at <destination> to seek a book near the Paradise Planets.  Return it to Alexandria for 80,000 credits."
	passengers 1
	source Alexandria
	destination
		near Talita 0 4
	to offer
		has "library card"
		not "alexandria seek active"
		or
			not "alexandria seeker level"
			and
				"alexandria seeker level" == 1
				"alexandria random" < 3
			and
				"alexandria seeker level" == 2
				"alexandria random" < 2
	on accept
		set "alexandria seek active"
		set "seek near paradise"
	on fail
		clear "alexandria seek active"
		clear "seek near paradise"
	on decline
		clear "alexandria seek active"
		clear "seek near paradise"


mission "Alexandria Seek Near Paradise Planets: Search"
	name "Alexandria Seek: <planet>"
	description "Continue to <destination> to seek a book near the Paradise Planets.  Return it to Alexandria for 80,000 credits."
	landing
	repeat
	name "Alexandria Seek: <planet>"
	source
		near Talita 0 7
	destination
		near Talita 0 7
		distance 1
		not
			attributes uninhabited
	to offer
		has "seek near paradise"
		"seek near paradise" < 5
	on offer
		conversation
			branch first
				"seek near paradise" == 1
			branch found
				random < 30
				"seek near paradise" > 1
			`It isn't here, but I think it may be on <destination>.`
				goto choice
			label found
			apply
				"seek near paradise" = 5
			`It isn't here, but I think it may be on <destination>.`
				goto choice
			label first
			`Thank you for meeting me.  We'll start searching at <destination>.`
			label choice
			choice
				`"I'll take you to <destination>."`
					accept
				`"I'm sorry, but I don't have time to finish this Seek.  You will have to find another captain."`
					decline
	on fail
		clear "alexandria seek active"
		clear "seek near paradise"
	on decline
		clear "alexandria seek active"
		clear "seek near paradise"
	on complete
		"seek near paradise" ++

mission "Alexandria Seek Near Paradise Planets: Return"
	name "Alexandria Seek: <planet>"
	description "Return the book to Alexandria for 80,000 credits."
	landing
	repeat
	destination Alexandria
	source
		near Talita 0 7
	to offer
		has "alexandria seek active"
		"seek near paradise" == 5
	on offer
		"seek near paradise" ++
		conversation
			`I have the book.  Let's take it back to the library at <destination> and get paid.`
			choice
				`"I'll take you to <destination>."`
				`"I'm sorry, but I don't have time to finish this Seek.  You will have to find another captain."`
	on fail
		clear "alexandria seek active"
		clear "seek near paradise"
	on decline
		clear "alexandria seek active"
		clear "seek near paradise"
	on complete
		payment 80000
		clear "alexandria seek active"
		clear "seek near paradise"
