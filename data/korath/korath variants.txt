# Low-tech Dredger for early boss fight. This is essentially a durable
# Korath Raider (Crippled). All outfits can be obtained from ships the
# player has already seen. Extra cooling and power will make ion and
# heat damage less effective. Fewer rifles than usual so you have a
# better chance of capturing it (though it isn't an ideal catch due to
# the missing nanite factory).

	add attributes
		# Remove the built-in nanite factory:
		"ramscoop" -2
		"shield generation" -2.5
		"shield energy" -3.4
		"hull repair rate" -1.6
		"hull energy" -1.6
		"hull heat" -2.1
		"energy capacity" -1800

		"Korath Grab-Strike" 2

		"Small Heat Shunt"

		"Steering (Planetary Class)"
